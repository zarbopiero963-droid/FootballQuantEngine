import os
import requests


BASE_URL = "https://v3.football.api-sports.io"


class APIFootballClient:

    def __init__(self, api_key=None):
        self.api_key = api_key or os.getenv("API_FOOTBALL_KEY")

        if not self.api_key:
            raise RuntimeError("API_FOOTBALL_KEY missing")

        self.headers = {
            "x-apisports-key": self.api_key,
        }

    def _get(self, path, params=None):
        url = BASE_URL + path

        response = requests.get(
            url,
            headers=self.headers,
            params=params,
            timeout=30,
        )
        response.raise_for_status()

        payload = response.json()

        if "response" not in payload:
            raise RuntimeError("API-Football response missing 'response' field")

        return payload["response"]

    def get_leagues(self):
        return self._get("/leagues")

    def get_fixtures(self, league, season):
        data = self._get(
            "/fixtures",
            {
                "league": league,
                "season": season,
            },
        )

        fixtures = []

        for item in data:
            fixtures.append(
                {
                    "fixture_id": str(item["fixture"]["id"]),
                    "home_team": item["teams"]["home"]["name"],
                    "away_team": item["teams"]["away"]["name"],
                }
            )

        return fixtures

    def get_completed_matches(self, league, season):
        data = self._get(
            "/fixtures",
            {
                "league": league,
                "season": season,
                "status": "FT",
            },
        )

        matches = []

        for item in data:
            home_goals = item.get("goals", {}).get("home")
            away_goals = item.get("goals", {}).get("away")

            if home_goals is None or away_goals is None:
                continue

            matches.append(
                {
                    "fixture_id": str(item["fixture"]["id"]),
                    "home_team": item["teams"]["home"]["name"],
                    "away_team": item["teams"]["away"]["name"],
                    "home_goals": int(home_goals),
                    "away_goals": int(away_goals),
                }
            )

        return matches

    def get_odds(self, fixture_ids):
        odds_map = {}

        for fixture_id in fixture_ids:
            data = self._get(
                "/odds",
                {
                    "fixture": fixture_id,
                },
            )

            if not data:
                continue

            try:
                bookmakers = data[0].get("bookmakers", [])
                if not bookmakers:
                    continue

                bets = bookmakers[0].get("bets", [])

                odds = {}

                for bet in bets:
                    if bet.get("name") != "Match Winner":
                        continue

                    for value in bet.get("values", []):
                        label = value.get("value")
                        odd = value.get("odd")

                        if odd in (None, ""):
                            continue

                        if label == "Home":
                            odds["home"] = float(odd)
                        elif label == "Draw":
                            odds["draw"] = float(odd)
                        elif label == "Away":
                            odds["away"] = float(odd)

                if odds:
                    odds_map[str(fixture_id)] = odds

            except Exception:
                continue

        return odds_map

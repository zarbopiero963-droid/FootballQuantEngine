CREA_FILE tests/exe/test_installed_exe_smoke.py
import os
import subprocess
import sys
import time

import pytest

pytestmark = [pytest.mark.exe, pytest.mark.windows]

if not sys.platform.startswith("win"):
    pytest.skip("EXE smoke tests run only on Windows", allow_module_level=True)

psutil = pytest.importorskip("psutil")


EXE_PATH = os.environ.get(
    "APP_EXE_PATH",
    r"dist\FootballQuantEngine\FootballQuantEngine.exe",
)


def test_installed_exe_starts_process():
    if not os.path.exists(EXE_PATH):
        pytest.skip(f"EXE not found: {EXE_PATH}")

    proc = subprocess.Popen(
        [EXE_PATH],
        stdout=subprocess.DEVNULL,
        stderr=subprocess.DEVNULL,
        stdin=subprocess.DEVNULL,
    )

    try:
        time.sleep(5)
        assert psutil.pid_exists(proc.pid)
    finally:
        try:
            proc.terminate()
            proc.wait(timeout=10)
        except Exception:
            try:
                proc.kill()
            except Exception:
                pass


def test_installed_exe_does_not_exit_immediately():
    if not os.path.exists(EXE_PATH):
        pytest.skip(f"EXE not found: {EXE_PATH}")

    proc = subprocess.Popen(
        [EXE_PATH],
        stdout=subprocess.DEVNULL,
        stderr=subprocess.DEVNULL,
        stdin=subprocess.DEVNULL,
    )

    try:
        time.sleep(3)
        assert proc.poll() is None
    finally:
        try:
            proc.terminate()
            proc.wait(timeout=10)
        except Exception:
            try:
                proc.kill()
            except Exception:
                pass
EOF

CREA_FILE tests/ui/test_main_window_qt.py
import pytest

pytestmark = [pytest.mark.ui]

pytest.importorskip("PySide6")
pytest.importorskip("pytestqt")

try:
    from PySide6.QtCore import Qt
    from ui.main_window import MainWindow
except Exception as exc:
    pytest.skip(
        f"Qt UI not available in this environment: {exc}",
        allow_module_level=True,
    )


class DummyController:
    def _results(self):
        return [
            {
                "fixture_id": "1",
                "home_team": "TeamA",
                "away_team": "TeamB",
                "market": "home",
                "probability": 0.58,
                "fair_odds": 1.72,
                "bookmaker_odds": 2.05,
                "market_edge": 0.06,
                "model_edge": 0.18,
                "confidence": 0.72,
                "agreement": 0.68,
                "decision": "BET",
            }
        ]

    def run_once(self):
        return self._results()

    def run_predictions(self):
        return self._results()


def test_ui_smoke(qtbot):
    window = MainWindow(DummyController())
    qtbot.addWidget(window)
    window.show()
    qtbot.waitUntil(lambda: window.isVisible(), timeout=3000)
    qtbot.mouseClick(window.run_button, Qt.LeftButton)
    assert window.dashboard is not None
EOF

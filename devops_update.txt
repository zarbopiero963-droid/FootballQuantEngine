from __future__ import annotations

import json
import os
import sqlite3
from contextlib import contextmanager


class RuntimeDB:

    def __init__(self, db_path: str = "outputs/runtime/runtime.db"):
        self.db_path = db_path
        os.makedirs(os.path.dirname(self.db_path), exist_ok=True)

    def connect(self) -> sqlite3.Connection:
        conn = sqlite3.connect(self.db_path, check_same_thread=False)
        conn.row_factory = sqlite3.Row
        conn.execute("PRAGMA journal_mode=WAL;")
        conn.execute("PRAGMA synchronous=NORMAL;")
        conn.execute("PRAGMA foreign_keys=ON;")
        conn.execute("PRAGMA busy_timeout=30000;")
        return conn

    @contextmanager
    def cursor(self):
        conn = self.connect()
        cur = conn.cursor()
        try:
            yield conn, cur
            conn.commit()
        except Exception:
            conn.rollback()
            raise
        finally:
            cur.close()
            conn.close()

    def init_db(self) -> None:
        with self.cursor() as (_, cur):
            cur.execute(
                """
                CREATE TABLE IF NOT EXISTS signals (
                    id INTEGER PRIMARY KEY AUTOINCREMENT,
                    fixture_id TEXT NOT NULL,
                    home_team TEXT,
                    away_team TEXT,
                    league TEXT,
                    market TEXT NOT NULL,
                    probability REAL,
                    fair_odds REAL,
                    bookmaker_odds REAL,
                    opening_odds REAL,
                    ev REAL,
                    confidence REAL,
                    agreement REAL,
                    decision TEXT,
                    suggested_stake REAL,
                    status TEXT DEFAULT 'OPEN',
                    created_at TEXT,
                    extra_json TEXT
                )
                """
            )

            cur.execute(
                """
                CREATE TABLE IF NOT EXISTS quotes (
                    id INTEGER PRIMARY KEY AUTOINCREMENT,
                    fixture_id TEXT NOT NULL,
                    market TEXT NOT NULL,
                    quote_type TEXT NOT NULL,
                    odds REAL,
                    source TEXT,
                    captured_at TEXT,
                    extra_json TEXT
                )
                """
            )

            cur.execute(
                """
                CREATE TABLE IF NOT EXISTS results (
                    id INTEGER PRIMARY KEY AUTOINCREMENT,
                    fixture_id TEXT NOT NULL,
                    market TEXT NOT NULL,
                    result TEXT,
                    pnl REAL,
                    closing_odds REAL,
                    clv_abs REAL,
                    clv_pct REAL,
                    settled_at TEXT,
                    extra_json TEXT
                )
                """
            )

            cur.execute(
                """
                CREATE TABLE IF NOT EXISTS scheduler_jobs (
                    id INTEGER PRIMARY KEY AUTOINCREMENT,
                    job_name TEXT UNIQUE NOT NULL,
                    interval_seconds REAL,
                    enabled INTEGER DEFAULT 1,
                    last_run_at TEXT,
                    next_run_at TEXT,
                    status TEXT DEFAULT 'idle',
                    last_error TEXT,
                    updated_at TEXT
                )
                """
            )

            cur.execute(
                """
                CREATE TABLE IF NOT EXISTS worker_state (
                    id INTEGER PRIMARY KEY AUTOINCREMENT,
                    worker_name TEXT UNIQUE NOT NULL,
                    status TEXT,
                    last_heartbeat TEXT,
                    current_job TEXT,
                    last_error TEXT,
                    updated_at TEXT
                )
                """
            )

            cur.execute(
                """
                CREATE TABLE IF NOT EXISTS cycle_state (
                    id INTEGER PRIMARY KEY CHECK (id = 1),
                    cycle_name TEXT,
                    status TEXT,
                    last_started_at TEXT,
                    last_completed_at TEXT,
                    last_success_at TEXT,
                    last_error TEXT,
                    payload_json TEXT
                )
                """
            )

            cur.execute(
                """
                CREATE TABLE IF NOT EXISTS final_reports (
                    id INTEGER PRIMARY KEY AUTOINCREMENT,
                    report_name TEXT,
                    report_path TEXT,
                    report_type TEXT,
                    created_at TEXT,
                    payload_json TEXT
                )
                """
            )

            cur.execute(
                """
                INSERT OR IGNORE INTO cycle_state (
                    id,
                    cycle_name,
                    status,
                    last_started_at,
                    last_completed_at,
                    last_success_at,
                    last_error,
                    payload_json
                ) VALUES (1, 'main_cycle', 'idle', NULL, NULL, NULL, NULL, NULL)
                """
            )

    def execute(self, sql: str, params: tuple = ()) -> None:
        with self.cursor() as (_, cur):
            cur.execute(sql, params)

    def executemany(self, sql: str, params_list: list[tuple]) -> None:
        with self.cursor() as (_, cur):
            cur.executemany(sql, params_list)

    def fetchall(self, sql: str, params: tuple = ()) -> list[dict]:
        with self.cursor() as (_, cur):
            cur.execute(sql, params)
            rows = cur.fetchall()
            return [dict(row) for row in rows]

    def fetchone(self, sql: str, params: tuple = ()) -> dict | None:
        with self.cursor() as (_, cur):
            cur.execute(sql, params)
            row = cur.fetchone()
            return dict(row) if row else None

    def json_dumps(self, payload: dict | list | None) -> str | None:
        if payload is None:
            return None
        return json.dumps(payload, ensure_ascii=False)

from __future__ import annotations

from datetime import datetime

from quant.runtime.runtime_db import RuntimeDB


class SignalStore:

    def __init__(self, db: RuntimeDB | None = None):
        self.db = db or RuntimeDB()
        self.db.init_db()

    def save_signals(self, rows: list[dict]) -> int:
        now = datetime.utcnow().isoformat() + "Z"

        payloads = []
        for row in rows:
            payloads.append(
                (
                    str(row.get("fixture_id", "")),
                    row.get("home_team", ""),
                    row.get("away_team", ""),
                    row.get("league", "UNKNOWN"),
                    row.get("market", ""),
                    float(row.get("probability", 0.0) or 0.0),
                    float(row.get("fair_odds", 0.0) or 0.0),
                    float(row.get("bookmaker_odds", 0.0) or 0.0),
                    float(row.get("opening_odds", row.get("bookmaker_odds", 0.0)) or 0.0),
                    float(row.get("ev", 0.0) or 0.0),
                    float(row.get("confidence", 0.0) or 0.0),
                    float(row.get("agreement", 0.0) or 0.0),
                    row.get("decision", "NO_BET"),
                    float(row.get("suggested_stake", 0.0) or 0.0),
                    row.get("status", "OPEN"),
                    now,
                    self.db.json_dumps(row),
                )
            )

        self.db.executemany(
            """
            INSERT INTO signals (
                fixture_id,
                home_team,
                away_team,
                league,
                market,
                probability,
                fair_odds,
                bookmaker_odds,
                opening_odds,
                ev,
                confidence,
                agreement,
                decision,
                suggested_stake,
                status,
                created_at,
                extra_json
            ) VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?)
            """,
            payloads,
        )

        return len(payloads)

    def list_signals(self, limit: int = 200) -> list[dict]:
        return self.db.fetchall(
            """
            SELECT *
            FROM signals
            ORDER BY id DESC
            LIMIT ?

from __future__ import annotations

from datetime import datetime

from quant.runtime.runtime_db import RuntimeDB


class QuoteStore:

    def __init__(self, db: RuntimeDB | None = None):
        self.db = db or RuntimeDB()
        self.db.init_db()

    def save_quote(
        self,
        fixture_id: str,
        market: str,
        quote_type: str,
        odds: float,
        source: str = "system",
        extra: dict | None = None,
    ) -> dict:
        captured_at = datetime.utcnow().isoformat() + "Z"

        self.db.execute(
            """
            INSERT INTO quotes (
                fixture_id,
                market,
                quote_type,
                odds,
                source,
                captured_at,
                extra_json
            ) VALUES (?, ?, ?, ?, ?, ?, ?)
            """,
            (
                str(fixture_id),
                str(market),
                str(quote_type),
                float(odds),
                str(source),
                captured_at,
                self.db.json_dumps(extra),
            ),
        )

        return {
            "fixture_id": str(fixture_id),
            "market": str(market),
            "quote_type": str(quote_type),
            "odds": float(odds),
            "source": str(source),
            "captured_at": captured_at,
        }

    def save_opening_quotes_from_rows(self, rows: list[dict], source: str = "value_engine") -> int:
        count = 0

        for row in rows:
            self.save_quote(
                fixture_id=str(row.get("fixture_id", "")),
                market=str(row.get("market", "")),
                quote_type="opening",
                odds=float(row.get("opening_odds", row.get("bookmaker_odds", 0.0)) or 0.0),
                source=source,
                extra=row,
            )
            count += 1

        return count

    def save_closing_quote(
        self,
        fixture_id: str,
        market: str,
        odds: float,
        source: str = "closing_collector",
        extra: dict | None = None,
    ) -> dict:
        return self.save_quote(
            fixture_id=fixture_id,
            market=market,
            quote_type="closing",
            odds=odds,
            source=source,
            extra=extra,
        )

    def list_quotes(self, limit: int = 500) -> list[dict]:
        return self.db.fetchall(
            """
            SELECT *
            FROM quotes
            ORDER BY id DESC
            LIMIT ?
            """,
            (int(limit),),
        )

from __future__ import annotations

from datetime import datetime

from quant.runtime.runtime_db import RuntimeDB


class JobRegistry:

    def __init__(self, db: RuntimeDB | None = None):
        self.db = db or RuntimeDB()
        self.db.init_db()

    def upsert_job(
        self,
        job_name: str,
        interval_seconds: float,
        enabled: bool = True,
        status: str = "idle",
    ) -> None:
        now = datetime.utcnow().isoformat() + "Z"

        existing = self.db.fetchone(
            "SELECT id FROM scheduler_jobs WHERE job_name = ?",
            (str(job_name),),
        )

        if existing:
            self.db.execute(
                """
                UPDATE scheduler_jobs
                SET interval_seconds = ?,
                    enabled = ?,
                    status = ?,
                    updated_at = ?
                WHERE job_name = ?
                """,
                (
                    float(interval_seconds),
                    1 if enabled else 0,
                    str(status),
                    now,
                    str(job_name),
                ),
            )
        else:
            self.db.execute(
                """
                INSERT INTO scheduler_jobs (
                    job_name,
                    interval_seconds,
                    enabled,
                    status,
                    updated_at
                ) VALUES (?, ?, ?, ?, ?)
                """,
                (
                    str(job_name),
                    float(interval_seconds),
                    1 if enabled else 0,
                    str(status),
                    now,
                ),
            )

    def mark_started(self, job_name: str) -> None:
        now = datetime.utcnow().isoformat() + "Z"
        self.db.execute(
            """
            UPDATE scheduler_jobs
            SET status = 'running',
                last_run_at = ?,
                updated_at = ?
            WHERE job_name = ?
            """,
            (now, now, str(job_name)),
        )

    def mark_finished(self, job_name: str, next_run_at: str | None = None) -> None:
        now = datetime.utcnow().isoformat() + "Z"
        self.db.execute(
            """
            UPDATE scheduler_jobs
            SET status = 'idle',
                next_run_at = ?,
                last_error = NULL,
                updated_at = ?
            WHERE job_name = ?
            """,
            (next_run_at, now, str(job_name)),
        )

    def mark_error(self, job_name: str, error: str) -> None:
        now = datetime.utcnow().isoformat() + "Z"
        self.db.execute(
            """
            UPDATE scheduler_jobs
            SET status = 'error',
                last_error = ?,
                updated_at = ?
            WHERE job_name = ?
            """,
            (str(error), now, str(job_name)),
        )

    def list_jobs(self) -> list[dict]:
        return self.db.fetchall(
            """
            SELECT *
            FROM scheduler_jobs
            ORDER BY job_name ASC
            """
        )

from __future__ import annotations

import json
import os

from quant.analytics.integrated_analytics import IntegratedAnalytics
from quant.clv.clv_journal_service import CLVJournalService
from quant.live.closing_odds_sync import ClosingOddsSync
from quant.ops.full_integration_runner import FullIntegrationRunner
from quant.runtime.cycle_state_store import CycleStateStore
from quant.runtime.job_registry import JobRegistry
from quant.runtime.quote_store import QuoteStore
from quant.runtime.report_store import ReportStore
from quant.runtime.result_store import ResultStore
from quant.runtime.runtime_db import RuntimeDB
from quant.runtime.signal_store import SignalStore
from quant.runtime.worker_state_store import WorkerStateStore


class UnifiedRuntime:

    def __init__(self):
        self.db = RuntimeDB()
        self.db.init_db()

        self.signal_store = SignalStore(self.db)
        self.quote_store = QuoteStore(self.db)
        self.result_store = ResultStore(self.db)
        self.job_registry = JobRegistry(self.db)
        self.worker_state = WorkerStateStore(self.db)
        self.cycle_state = CycleStateStore(self.db)
        self.report_store = ReportStore(self.db)

        self.integration_runner = FullIntegrationRunner()
        self.clv_service = CLVJournalService()
        self.closing_sync = ClosingOddsSync()
        self.analytics = IntegratedAnalytics()

        self.report_json_path = "outputs/runtime/final_runtime_report.json"
        os.makedirs(os.path.dirname(self.report_json_path), exist_ok=True)

    def run_full_cycle(
        self,
        bankroll: float = 1000.0,
        league: str = "Serie A",
        season: int = 2024,
    ) -> dict:
        self.worker_state.update_worker(
            worker_name="main_worker",
            status="running",
            current_job="full_cycle",
            last_error=None,
        )
        self.cycle_state.mark_started("full_cycle")

        try:
            payload = self.integration_runner.run_cycle(
                bankroll=bankroll,
                league=league,
                season=season,
            )

            ranked_rows = payload.get("ranked_rows", [])
            self.signal_store.save_signals(ranked_rows)
            self.quote_store.save_opening_quotes_from_rows(ranked_rows, source="full_integration")

            runtime_report = {
                "performance_summary": payload.get("performance_summary", {}),
                "threshold_summary": payload.get("threshold_summary", {}),
                "analytics": payload.get("analytics", {}),
                "temporal_guard": payload.get("temporal_guard", {}),
                "counts": {
                    "ranked_rows": len(ranked_rows),
                    "journal_rows": len(payload.get("journal_rows", [])),
                    "fusion_rows": len(payload.get("fusion", {}).get("rows", [])),
                },
            }

            with open(self.report_json_path, "w", encoding="utf-8") as handle:
                json.dump(runtime_report, handle, indent=2)

            self.report_store.save_report(
                report_name="final_runtime_report",
                report_path=self.report_json_path,
                report_type="json",
                payload=runtime_report,
            )

            self.cycle_state.mark_completed(runtime_report)
            self.worker_state.update_worker(
                worker_name="main_worker",
                status="idle",
                current_job=None,
                last_error=None,
            )

            return {
                "ok": True,
                "payload": payload,
                "runtime_report": runtime_report,
                "report_path": self.report_json_path,
            }

        except Exception as exc:
            self.cycle_state.mark_error(str(exc))
            self.worker_state.update_worker(
                worker_name="main_worker",
                status="error",
                current_job="full_cycle",
                last_error=str(exc),
            )
            raise

    def ingest_closing_and_settle(
        self,
        fixture_id: str,
        market: str,
        closing_odds: float,
        result: str,
    ) -> dict:
        sync_payload = self.closing_sync.ingest_and_sync(
            fixture_id=fixture_id,
            market=market,
            closing_odds=closing_odds,
            result=result,
        )

        journal_update = sync_payload.get("journal_update")
        if journal_update:
            self.quote_store.save_closing_quote(
                fixture_id=fixture_id,
                market=market,
                odds=closing_odds,
                source="closing_sync",
                extra=sync_payload,
            )

            self.result_store.save_result(
                fixture_id=fixture_id,
                market=market,
                result=journal_update.get("result", result),
                pnl=float(journal_update.get("pnl", 0.0) or 0.0),
                closing_odds=float(journal_update.get("closing_odds", closing_odds) or closing_odds),
                clv_abs=float(journal_update.get("clv_abs", 0.0) or 0.0),
                clv_pct=float(journal_update.get("clv_pct", 0.0) or 0.0),
                extra=journal_update,
            )

        return sync_payload

    def refresh_final_report(self) -> dict:
        journal_rows = self.clv_service.journal.list_bets()
        analytics_payload = self.analytics.build(journal_rows)
        performance_summary = self.clv_service.performance_summary()

        final_payload = {
            "performance_summary": performance_summary,
            "analytics": analytics_payload,
            "signals_count": len(self.signal_store.list_signals()),
            "quotes_count": len(self.quote_store.list_quotes()),
            "results_count": len(self.result_store.list_results()),
            "jobs": self.job_registry.list_jobs(),
            "workers": self.worker_state.list_workers(),
            "cycle_state": self.cycle_state.get_state(),
        }

        with open(self.report_json_path, "w", encoding="utf-8") as handle:
            json.dump(final_payload, handle, indent=2)

        self.report_store.save_report(
            report_name="final_runtime_report_refresh",
            report_path=self.report_json_path,
            report_type="json",
            payload=final_payload,
        )

        return {
            "ok": True,

CREA_FILE quant/runtime/orchestration_service.py
from __future__ import annotations

from quant.ops.scheduler_service import SchedulerService
from quant.ops.watchdog import Watchdog
from quant.ops.worker_pool import WorkerPool
from quant.runtime.unified_runtime import UnifiedRuntime


class OrchestrationService:

    def __init__(self):
        self.runtime = UnifiedRuntime()
        self.scheduler = SchedulerService()
        self.watchdog = Watchdog()
        self.worker_pool = WorkerPool(max_workers=2)

    def _cycle_job(self):
        self.watchdog.ping("orchestration_cycle")
        return self.runtime.run_full_cycle()

    def register_default_jobs(self, interval_seconds: float = 300.0) -> None:
        self.runtime.job_registry.upsert_job(
            job_name="full_cycle",
            interval_seconds=interval_seconds,
            enabled=True,
            status="idle",
        )

        self.scheduler.add_recurring_job(
            name="full_cycle",
            callback=lambda: self.worker_pool.submit(self._cycle_job),
            interval_seconds=interval_seconds,
            run_immediately=True,
        )

    def start(self, interval_seconds: float = 300.0) -> None:
        self.register_default_jobs(interval_seconds=interval_seconds)
        self.scheduler.start()

    def stop(self) -> None:
        self.scheduler.stop()
        self.worker_pool.shutdown(wait=False)

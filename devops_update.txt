import pytest

pytest.importorskip("PySide6")
pytest.importorskip("pytestqt")

try:
    from PySide6.QtCore import Qt
    from ui.main_window import MainWindow
except Exception as exc:
    pytest.skip(f"Qt UI not available in this environment: {exc}", allow_module_level=True)


class DummyController:
    def _results(self):
        return [
            {
                "match_id": "TeamA_vs_TeamB",
                "market": "home",
                "probability": 0.58,
                "odds": 2.05,
                "model_edge": 0.04,
                "market_edge": 0.06,
                "confidence": 0.72,
                "agreement": 0.68,
                "decision": "BET",
            },
            {
                "match_id": "TeamC_vs_TeamD",
                "market": "draw",
                "probability": 0.31,
                "odds": 3.10,
                "model_edge": 0.01,
                "market_edge": 0.02,
                "confidence": 0.55,
                "agreement": 0.52,
                "decision": "WATCHLIST",
            },
            {
                "match_id": "TeamE_vs_TeamF",
                "market": "away",
                "probability": 0.22,
                "odds": 4.20,
                "model_edge": -0.01,
                "market_edge": -0.02,
                "confidence": 0.34,
                "agreement": 0.44,
                "decision": "NO_BET",
            },
        ]

    def run_once(self):
        return self._results()

    def run_predictions(self):
        return self._results()


def build_window(qtbot):
    controller = DummyController()
    window = MainWindow(controller)
    qtbot.addWidget(window)
    window.show()
    qtbot.waitUntil(lambda: window.isVisible(), timeout=5000)
    return window


def test_main_window_opens(qtbot):
    window = build_window(qtbot)
    assert "Football Quant Engine" in window.windowTitle()


def test_run_cycle_populates_dashboard(qtbot):
    window = build_window(qtbot)

    qtbot.mouseClick(window.run_button, Qt.LeftButton)

    qtbot.waitUntil(
        lambda: window.dashboard.results_table.rowCount() == 3,
        timeout=5000,
    )

    assert window.dashboard.results_table.rowCount() == 3
    assert window.dashboard.bet_card.value_label.text() == "1"
    assert window.dashboard.watchlist_card.value_label.text() == "1"
    assert window.dashboard.no_bet_card.value_label.text() == "1"


def test_decision_filter_works(qtbot):
    window = build_window(qtbot)

    qtbot.mouseClick(window.run_button, Qt.LeftButton)

    qtbot.waitUntil(
        lambda: window.dashboard.results_table.rowCount() == 3,
        timeout=5000,
    )

    window.dashboard.decision_filter.setCurrentText("BET")

    qtbot.waitUntil(
        lambda: window.dashboard.results_table.rowCount() == 1,
        timeout=5000,
    )

    assert window.dashboard.results_table.rowCount() == 1


def test_market_filter_works(qtbot):

def test_open_secondary_windows(qtbot):
    window = build_window(qtbot)

    qtbot.mouseClick(window.settings_button, Qt.LeftButton)
    qtbot.waitUntil(lambda: window.settings_window is not None, timeout=3000)

    qtbot.mouseClick(window.about_button, Qt.LeftButton)
    qtbot.waitUntil(lambda: window.about_dialog is not None, timeout=3000)

    qtbot.mouseClick(window.project_summary_button, Qt.LeftButton)
    qtbot.waitUntil(lambda: window.project_summary_window is not None, timeout=3000)

    qtbot.mouseClick(window.final_check_button, Qt.LeftButton)
    qtbot.waitUntil(lambda: window.final_check_window is not None, timeout=3000)

    assert window.settings_window is not None
    assert window.about_dialog is not None
    assert window.project_summary_window is not None
    assert window.final_check_window is not None

from pathlib import Path


class RepoHealthCheck:

    def required_files(self):

        return [
            "README.md",
            "requirements.txt",
            "football_quant_engine.spec",
            ".github/workflows/build.yml",
            "database/schema.sql",
            "app/main.py",
            "app/controller.py",
            "engine/job_runner.py",
            "engine/prediction_pipeline.py",
            "engine/plugin_loader.py",
            "engine/offline_engine.py",
            "models/poisson_model.py",
            "models/bivariate_poisson.py",
            "models/elo_model.py",
            "models/bayesian_model.py",
            "analysis/probability_markets.py",
            "analysis/report_generator.py",
            "features/feature_engine.py",
            "features/offline_features.py",
            "training/dataset_builder.py",
            "training/backtest_engine.py",
            "training/backtest_metrics.py",
            "plugins/models/poisson_plugin.py",
            "plugins/models/elo_plugin.py",
            "plugins/models/bivariate_plugin.py",
        ]

    def missing_files(self):

        missing = []

        for filepath in self.required_files():
            if not Path(filepath).exists():
                missing.append(filepath)

        return missing

    def outputs_dir_exists(self):

        return Path("outputs").exists()

    def check(self):

        missing = self.missing_files()

        return {
            "ok": len(missing) == 0,
            "missing_files": missing,
            "outputs_dir_exists": self.outputs_dir_exists(),
        }

from pathlib import Path
import sqlite3

from config.constants import DATABASE_NAME


BASE_DIR = Path(__file__).resolve().parent
SCHEMA_PATH = BASE_DIR / "schema.sql"


def connect():

    return sqlite3.connect(DATABASE_NAME)


def init_db():

    conn = connect()

    with open(SCHEMA_PATH, encoding="utf-8") as f:
        conn.executescript(f.read())

    conn.commit()
    conn.close()

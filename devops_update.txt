from __future__ import annotations

import os
import sqlite3
from contextlib import contextmanager


class BootstrapDB:

    def __init__(self, db_path: str = "outputs/bootstrap/bootstrap.db"):
        self.db_path = db_path
        os.makedirs(os.path.dirname(self.db_path), exist_ok=True)

    def connect(self) -> sqlite3.Connection:
        conn = sqlite3.connect(self.db_path, check_same_thread=False)
        conn.row_factory = sqlite3.Row
        conn.execute("PRAGMA journal_mode=WAL;")
        conn.execute("PRAGMA synchronous=NORMAL;")
        conn.execute("PRAGMA foreign_keys=ON;")
        conn.execute("PRAGMA busy_timeout=30000;")
        return conn

    @contextmanager
    def cursor(self):
        conn = self.connect()
        cur = conn.cursor()
        try:
            yield conn, cur
            conn.commit()
        except Exception:
            conn.rollback()
            raise
        finally:
            cur.close()
            conn.close()

    def init_db(self) -> None:
        with self.cursor() as (_, cur):
            cur.execute(
                """
                CREATE TABLE IF NOT EXISTS bootstrap_state (
                    id INTEGER PRIMARY KEY CHECK (id = 1),
                    first_run_completed INTEGER DEFAULT 0,
                    bootstrap_started_at TEXT,
                    bootstrap_completed_at TEXT,
                    last_sync_at TEXT,
                    status TEXT DEFAULT 'idle',
                    last_error TEXT
                )
                """
            )

            cur.execute(
                """
                INSERT OR IGNORE INTO bootstrap_state (
                    id,
                    first_run_completed,
                    bootstrap_started_at,
                    bootstrap_completed_at,
                    last_sync_at,
                    status,
                    last_error
                ) VALUES (1, 0, NULL, NULL, NULL, 'idle', NULL)
                """
            )

            cur.execute(
                """
                CREATE TABLE IF NOT EXISTS fixtures (
                    fixture_id TEXT PRIMARY KEY,
                    country TEXT,
                    league TEXT,
                    season INTEGER,
                    home_team TEXT,
                    away_team TEXT,
                    match_date TEXT,
                    status TEXT,
                    is_live INTEGER DEFAULT 0,
                    is_upcoming INTEGER DEFAULT 0,
                    source TEXT,
                    extra_json TEXT,
                    updated_at TEXT
                )
                """
            )

            cur.execute(
                """
                CREATE TABLE IF NOT EXISTS fixture_quotes (
                    id INTEGER PRIMARY KEY AUTOINCREMENT,
                    fixture_id TEXT NOT NULL,
                    market TEXT NOT NULL,
                    odds REAL,
                    quote_type TEXT NOT NULL,
                    source TEXT,
                    captured_at TEXT,
                    extra_json TEXT
                )
                """
            )

            cur.execute(
                """
                CREATE TABLE IF NOT EXISTS fixture_features (
                    fixture_id TEXT PRIMARY KEY,
                    country TEXT,
                    league TEXT,
                    season INTEGER,
                    xg_diff REAL,
                    xga_diff REAL,
                    xpts_diff REAL,
                    shots_diff REAL,
                    shots_on_target_diff REAL,
                    form_score_diff REAL,
                    bookmaker_odds_home REAL,
                    bookmaker_odds_draw REAL,
                    bookmaker_odds_away REAL,
                    extra_json TEXT,
                    updated_at TEXT
                )
                """
            )

    def execute(self, sql: str, params: tuple = ()) -> None:
        with self.cursor() as (_, cur):
            cur.execute(sql, params)

    def fetchone(self, sql: str, params: tuple = ()) -> dict | None:
        with self.cursor() as (_, cur):
            cur.execute(sql, params)
            row = cur.fetchone()
            return dict(row) if row else None

    def fetchall(self, sql: str, params: tuple = ()) -> list[dict]:
        with self.cursor() as (_, cur):
            cur.execute(sql, params)
            rows = cur.fetchall()
            return [dict(row) for row in rows]

from __future__ import annotations

from datetime import datetime

from quant.bootstrap.bootstrap_db import BootstrapDB


class BootstrapStateStore:

    def __init__(self, db: BootstrapDB | None = None):
        self.db = db or BootstrapDB()
        self.db.init_db()

    def is_first_run_completed(self) -> bool:
        row = self.db.fetchone(
            "SELECT first_run_completed FROM bootstrap_state WHERE id = 1"
        )
        return bool(row and int(row.get("first_run_completed", 0)) == 1)

    def mark_started(self) -> None:
        now = datetime.utcnow().isoformat() + "Z"
        self.db.execute(
            """
            UPDATE bootstrap_state
            SET bootstrap_started_at = ?,
                status = 'running',
                last_error = NULL
            WHERE id = 1
            """,
            (now,),
        )

    def mark_completed(self) -> None:
        now = datetime.utcnow().isoformat() + "Z"
        self.db.execute(
            """
            UPDATE bootstrap_state
            SET first_run_completed = 1,
                bootstrap_completed_at = ?,
                last_sync_at = ?,
                status = 'completed',
                last_error = NULL
            WHERE id = 1
            """,
            (now, now),
        )

    def mark_sync(self) -> None:
        now = datetime.utcnow().isoformat() + "Z"
        self.db.execute(
            """
            UPDATE bootstrap_state
            SET last_sync_at = ?,
                status = 'completed',
                last_error = NULL
            WHERE id = 1
            """,
            (now,),
        )

    def mark_error(self, error: str) -> None:
        self.db.execute(
            """
            UPDATE bootstrap_state
            SET status = 'error',
                last_error = ?
            WHERE id = 1
            """,
            (str(error),),
        )

    def get_state(self) -> dict | None:
        return self.db.fetchone("SELECT * FROM bootstrap_state WHERE id = 1")

from __future__ import annotations

import json
from datetime import datetime

from quant.bootstrap.bootstrap_db import BootstrapDB


class FixtureStore:

    def __init__(self, db: BootstrapDB | None = None):
        self.db = db or BootstrapDB()
        self.db.init_db()

    def upsert_fixture(self, row: dict, source: str = "bootstrap") -> None:
        now = datetime.utcnow().isoformat() + "Z"

        self.db.execute(
            """
            INSERT OR REPLACE INTO fixtures (
                fixture_id,
                country,
                league,
                season,
                home_team,
                away_team,
                match_date,
                status,
                is_live,
                is_upcoming,
                source,
                extra_json,
                updated_at
            ) VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?)
            """,
            (
                str(row.get("fixture_id", "")),
                row.get("country", "UNKNOWN"),
                row.get("league", "UNKNOWN"),
                int(row.get("season", 0) or 0),
                row.get("home_team", ""),
                row.get("away_team", ""),
                row.get("match_date", ""),
                row.get("status", "NS"),
                1 if bool(row.get("is_live", False)) else 0,
                1 if bool(row.get("is_upcoming", False)) else 0,
                source,
                json.dumps(row, ensure_ascii=False),
                now,
            ),
        )

    def upsert_many(self, rows: list[dict], source: str = "bootstrap") -> int:
        for row in rows:
            self.upsert_fixture(row, source=source)
        return len(rows)

    def count_fixtures(self) -> int:
        row = self.db.fetchone("SELECT COUNT(*) AS c FROM fixtures")
        return int(row["c"]) if row else 0

    def list_fixtures(self, limit: int = 200) -> list[dict]:
        return self.db.fetchall(
            """
            SELECT *
            FROM fixtures
            ORDER BY updated_at DESC
            LIMIT ?
            """,
            (int(limit),),

from __future__ import annotations

import json
from datetime import datetime

from quant.bootstrap.bootstrap_db import BootstrapDB


class QuoteStoreBootstrap:

    def __init__(self, db: BootstrapDB | None = None):
        self.db = db or BootstrapDB()
        self.db.init_db()

    def save_quote(
        self,
        fixture_id: str,
        market: str,
        odds: float,
        quote_type: str,
        source: str = "bootstrap",
        extra: dict | None = None,
    ) -> None:
        now = datetime.utcnow().isoformat() + "Z"

        self.db.execute(
            """
            INSERT INTO fixture_quotes (
                fixture_id,
                market,
                odds,
                quote_type,
                source,
                captured_at,
                extra_json
            ) VALUES (?, ?, ?, ?, ?, ?, ?)
            """,
            (
                str(fixture_id),
                str(market),
                float(odds or 0.0),
                str(quote_type),
                str(source),
                now,
                json.dumps(extra or {}, ensure_ascii=False),
            ),
        )

    def save_opening_quotes_from_fixture(self, row: dict, source: str = "bootstrap") -> None:
        fixture_id = str(row.get("fixture_id", ""))

        odds_map = {
            "home": float(row.get("bookmaker_odds_home", 0.0) or 0.0),
            "draw": float(row.get("bookmaker_odds_draw", 0.0) or 0.0),
            "away": float(row.get("bookmaker_odds_away", 0.0) or 0.0),
            "over_25": float(row.get("over_25_odds", 0.0) or 0.0),
            "under_25": float(row.get("under_25_odds", 0.0) or 0.0),
            "btts_yes": float(row.get("btts_yes_odds", 0.0) or 0.0),
            "btts_no": float(row.get("btts_no_odds", 0.0) or 0.0),
        }

        for market, odds in odds_map.items():
            if odds > 0.0:
                self.save_quote(
                    fixture_id=fixture_id,
                    market=market,
                    odds=odds,
                    quote_type="opening",
                    source=source,
                    extra=row,
                )

    def list_quotes(self, limit: int = 500) -> list[dict]:
        return self.db.fetchall(
            """
            SELECT *
            FROM fixture_quotes
            ORDER BY id DESC
            LIMIT ?
            """,
            (int(limit),),
        )

from __future__ import annotations

from quant.bootstrap.bootstrap_reporter import BootstrapReporter
from quant.bootstrap.bootstrap_state_store import BootstrapStateStore
from quant.bootstrap.feature_store_bootstrap import FeatureStoreBootstrap
from quant.bootstrap.fixture_store import FixtureStore
from quant.bootstrap.league_catalog import STANDARD_LEAGUES
from quant.bootstrap.quote_store_bootstrap import QuoteStoreBootstrap
from quant.bootstrap.sample_live_bootstrap_provider import SampleLiveBootstrapProvider


class InitialDBBootstrap:

    def __init__(self, provider=None):
        self.provider = provider or SampleLiveBootstrapProvider()
        self.state = BootstrapStateStore()
        self.fixtures = FixtureStore()
        self.quotes = QuoteStoreBootstrap()
        self.features = FeatureStoreBootstrap()
        self.reporter = BootstrapReporter()

    def _seed_league(self, country: str, league: str, season: int) -> dict:
        historical = self.provider.get_historical_matches(country, league, season)
        live = self.provider.get_live_matches(country, league, season)
        upcoming = self.provider.get_upcoming_matches(country, league, season)

        all_rows = historical + live + upcoming

        self.fixtures.upsert_many(all_rows, source="initial_bootstrap")
        self.features.upsert_many(all_rows)

        for row in all_rows:
            self.quotes.save_opening_quotes_from_fixture(row, source="initial_bootstrap")

        return {
            "country": country,
            "league": league,
            "season": season,
            "historical_count": len(historical),
            "live_count": len(live),
            "upcoming_count": len(upcoming),
            "total_count": len(all_rows),
        }

    def run_first_bootstrap(self) -> dict:
        self.state.mark_started()

        try:
            league_reports = []
            for item in STANDARD_LEAGUES:
                league_reports.append(
                    self._seed_league(
                        country=item["country"],
                        league=item["league"],
                        season=int(item["season"]),
                    )
                )

            self.state.mark_completed()

            payload = {
                "mode": "first_bootstrap",
                "first_run_completed": True,
                "league_reports": league_reports,
                "fixtures_count": self.fixtures.count_fixtures(),
                "quotes_count": len(self.quotes.list_quotes()),
                "features_count": len(self.features.list_features()),
                "state": self.state.get_state(),
            }
            report_path = self.reporter.save(payload)

            return {
                "ok": True,
                "payload": payload,
                "report_path": report_path,
            }

        except Exception as exc:
            self.state.mark_error(str(exc))
            raise

    def run_incremental_sync(self) -> dict:
        league_reports = []

        try:
            for item in STANDARD_LEAGUES:
                country = item["country"]
                league = item["league"]
                season = int(item["season"])

                live = self.provider.get_live_matches(country, league, season)
                upcoming = self.provider.get_upcoming_matches(country, league, season)

                delta_rows = live + upcoming

                self.fixtures.upsert_many(delta_rows, source="incremental_sync")
                self.features.upsert_many(delta_rows)

                for row in delta_rows:
                    self.quotes.save_opening_quotes_from_fixture(row, source="incremental_sync")

                league_reports.append(
                    {
                        "country": country,
                        "league": league,
                        "season": season,
                        "live_count": len(live),
                        "upcoming_count": len(upcoming),
                        "delta_count": len(delta_rows),
                    }
                )

            self.state.mark_sync()

            payload = {
                "mode": "incremental_sync",
                "league_reports": league_reports,
                "fixtures_count": self.fixtures.count_fixtures(),
                "quotes_count": len(self.quotes.list_quotes()),
                "features_count": len(self.features.list_features()),
                "state": self.state.get_state(),
            }
            report_path = self.reporter.save(payload)

            return {
                "ok": True,
                "payload": payload,
                "report_path": report_path,
            }

        except Exception as exc:
            self.state.mark_error(str(exc))
            raise

    def run(self) -> dict:
        if not self.state.is_first_run_completed():
            return self.run_first_bootstrap()

        return self.run_incremental_sync()

from __future__ import annotations

from quant.bootstrap.initial_db_bootstrap import InitialDBBootstrap
from quant.ops.scheduler_service import SchedulerService
from quant.ops.watchdog import Watchdog
from quant.ops.worker_pool import WorkerPool


class BootstrapScheduler:

    def __init__(self):
        self.bootstrap = InitialDBBootstrap()
        self.scheduler = SchedulerService()
        self.watchdog = Watchdog()
        self.worker_pool = WorkerPool(max_workers=2)

    def _run_cycle(self):
        self.watchdog.ping("bootstrap_scheduler")
        return self.bootstrap.run()

    def start(self, interval_seconds: float = 300.0):
        self.scheduler.add_recurring_job(
            name="bootstrap_live_sync",
            callback=lambda: self.worker_pool.submit(self._run_cycle),
            interval_seconds=interval_seconds,
            run_immediately=True,
        )
        self.scheduler.start()

    def stop(self):
        self.scheduler.stop()
        self.worker_pool.shutdown(wait=False)

from __future__ import annotations

import time

from quant.ops.heartbeat import Heartbeat
from quant.ops.monitoring import Monitoring


class Watchdog:

    def __init__(self, stale_after_seconds: int = 120):
        self.stale_after_seconds = int(stale_after_seconds)
        self.heartbeat = Heartbeat()
        self.monitoring = Monitoring()

    def ping(self, component: str):
        path = self.heartbeat.beat(component, "running")
        self.monitoring.log(
            "watchdog_ping",
            {
                "component": component,
                "heartbeat_path": path,
            },
        )
        return path

    def loop_ping(
        self,
        component: str,
        interval_seconds: float = 30.0,
        cycles: int = 1,
    ):
        for _ in range(max(1, int(cycles))):
            self.ping(component)
            time.sleep(float(interval_seconds))

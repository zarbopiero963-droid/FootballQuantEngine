from __future__ import annotations

import pandas as pd


class MLDatasetBuilder:

    FEATURE_COLUMNS = [
        "elo_diff",
        "form_diff",
        "xg_diff",
        "xga_diff",
        "xpts_diff",
        "shots_diff",
        "shots_on_target_diff",
        "form_score_diff",
        "bookmaker_odds_home",
        "bookmaker_odds_draw",
        "bookmaker_odds_away",
    ]

    def build_binary_home_dataset(self, rows: list[dict]) -> pd.DataFrame:
        if not rows:
            return pd.DataFrame(columns=self.FEATURE_COLUMNS + ["target"])

        df = pd.DataFrame(rows).copy()

        for col in self.FEATURE_COLUMNS:
            if col not in df.columns:
                df[col] = 0.0

        if "result" not in df.columns:
            df["result"] = "D"

        df["target"] = (df["result"].astype(str) == "H").astype(int)

        return df[self.FEATURE_COLUMNS + ["target"]].copy()

CREA_FILE quant/ml/probability_calibrator_real.py
from __future__ import annotations

import numpy as np

try:
    from sklearn.isotonic import IsotonicRegression
except Exception:
    IsotonicRegression = None


class ProbabilityCalibratorReal:

    def __init__(self):
        self.model = None
        self.is_fitted = False

    def fit(self, raw_probs, y_true):
        raw_probs = np.asarray(raw_probs, dtype=float)
        y_true = np.asarray(y_true, dtype=float)

        if len(raw_probs) == 0 or len(y_true) == 0 or len(raw_probs) != len(y_true):
            self.is_fitted = False
            self.model = None
            return

        if IsotonicRegression is None:
            self.is_fitted = False
            self.model = None
            return

        self.model = IsotonicRegression(out_of_bounds="clip")
        self.model.fit(raw_probs, y_true)
        self.is_fitted = True

    def transform(self, raw_probs):
        raw_probs = np.asarray(raw_probs, dtype=float)

        if not self.is_fitted or self.model is None:
            return np.clip(raw_probs, 0.0001, 0.9999)

        calibrated = self.model.predict(raw_probs)
        return np.clip(calibrated, 0.0001, 0.9999)

    def transform_one(self, raw_prob: float) -> float:
        return float(self.transform([raw_prob])[0])

CREA_FILE quant/ml/temporal_validator_real.py
from __future__ import annotations

import pandas as pd


class TemporalValidatorReal:

    def validate(self, df: pd.DataFrame, date_col: str = "match_date") -> list[dict]:
        if df is None or df.empty:
            return []

        work = df.copy()

        if date_col not in work.columns:
            return []

        work[date_col] = pd.to_datetime(work[date_col], errors="coerce")
        work = work.dropna(subset=[date_col]).sort_values(date_col).reset_index(drop=True)

        years = sorted(work[date_col].dt.year.unique().tolist())
        windows = []

        for idx in range(2, len(years)):
            train_years = years[:idx]
            validation_year = years[idx - 1]
            test_year = years[idx]

            train_df = work[work[date_col].dt.year.isin(train_years)].copy()
            validation_df = work[work[date_col].dt.year == validation_year].copy()
            test_df = work[work[date_col].dt.year == test_year].copy()

            if train_df.empty or validation_df.empty or test_df.empty:
                continue

            windows.append(
                {
                    "train_years": train_years,
                    "validation_year": validation_year,
                    "test_year": test_year,
                    "train_rows": len(train_df.index),
                    "validation_rows": len(validation_df.index),
                    "test_rows": len(test_df.index),
                }
            )

        return windows

CREA_FILE quant/ml/threshold_optimizer.py
from __future__ import annotations


class ThresholdOptimizer:

    def optimize(
        self,
        rows: list[dict],
        probability_key: str = "probability",
        ev_key: str = "ev",
        confidence_key: str = "confidence",
        agreement_key: str = "agreement",
    ) -> dict:
        if not rows:
            return {
                "min_probability": 0.54,
                "min_ev": 0.02,
                "min_confidence": 0.60,
                "min_agreement": 0.55,
            }

        best = None

        for p in (0.50, 0.52, 0.54, 0.56, 0.58):
            for ev in (0.01, 0.015, 0.02, 0.03):
                for conf in (0.55, 0.60, 0.65, 0.70):
                    for agree in (0.50, 0.55, 0.60, 0.65):
                        selected = [
                            row for row in rows
                            if float(row.get(probability_key, 0.0)) >= p
                            and float(row.get(ev_key, 0.0)) >= ev
                            and float(row.get(confidence_key, 0.0)) >= conf
                            and float(row.get(agreement_key, 0.0)) >= agree
                        ]

                        score = sum(float(row.get(ev_key, 0.0)) for row in selected)

                        candidate = {
                            "min_probability": p,
                            "min_ev": ev,
                            "min_confidence": conf,
                            "min_agreement": agree,
                            "selected_count": len(selected),
                            "score": round(score, 6),
                        }

                        if best is None or candidate["score"] > best["score"]:
                            best = candidate

        return best or {
            "min_probability": 0.54,
            "min_ev": 0.02,
            "min_confidence": 0.60,
            "min_agreement": 0.55,
        }

CREA_FILE quant/ml/ml_layer_real.py
from __future__ import annotations

import numpy as np
import pandas as pd

try:
    from sklearn.linear_model import LogisticRegression
except Exception:
    LogisticRegression = None

from quant.ml.ml_dataset import MLDatasetBuilder
from quant.ml.probability_calibrator_real import ProbabilityCalibratorReal


class MLLayerReal:

    def __init__(self):
        self.dataset_builder = MLDatasetBuilder()
        self.calibrator = ProbabilityCalibratorReal()
        self.model = None
        self.feature_columns = list(self.dataset_builder.FEATURE_COLUMNS)
        self.is_fitted = False

    def fit(self, rows: list[dict]):
        df = self.dataset_builder.build_binary_home_dataset(rows)

        if df.empty or LogisticRegression is None:
            self.model = None
            self.is_fitted = False
            return

        X = df[self.feature_columns].astype(float).values
        y = df["target"].astype(int).values

        if len(set(y.tolist())) < 2:
            self.model = None
            self.is_fitted = False
            return

        self.model = LogisticRegression(max_iter=1000)
        self.model.fit(X, y)

        raw_probs = self.model.predict_proba(X)[:, 1]
        self.calibrator.fit(raw_probs, y)

        self.is_fitted = True

    def predict_home_probability(self, row: dict) -> float:
        if not self.is_fitted or self.model is None:
            return 0.50

        values = []
        for col in self.feature_columns:
            values.append(float(row.get(col, 0.0) or 0.0))

        X = np.asarray([values], dtype=float)
        raw_prob = float(self.model.predict_proba(X)[0, 1])
        calibrated = self.calibrator.transform_one(raw_prob)

        return round(float(calibrated), 6)

from __future__ import annotations

import json
import os
from datetime import datetime


class ClosingOddsCollector:

    def __init__(self, path: str = "outputs/ops/closing_odds.json"):
        self.path = path
        os.makedirs(os.path.dirname(self.path), exist_ok=True)

        if not os.path.exists(self.path):
            with open(self.path, "w", encoding="utf-8") as handle:
                json.dump([], handle, indent=2)

    def _load(self) -> list[dict]:
        with open(self.path, "r", encoding="utf-8") as handle:
            return json.load(handle)

    def _save(self, rows: list[dict]) -> None:
        with open(self.path, "w", encoding="utf-8") as handle:
            json.dump(rows, handle, indent=2)

    def collect(self, fixture_id: str, market: str, closing_odds: float) -> dict:
        rows = self._load()

        row = {
            "fixture_id": str(fixture_id),
            "market": str(market),
            "closing_odds": float(closing_odds),
            "collected_at": datetime.utcnow().isoformat() + "Z",
        }

        rows.append(row)
        self._save(rows)
        return row

    def list_rows(self) -> list[dict]:
        return self._load()

from __future__ import annotations

import pandas as pd


class LeaguePerformanceAnalyzer:

    def analyze(self, rows: list[dict]) -> list[dict]:
        if not rows:
            return []

        df = pd.DataFrame(rows).copy()

        if "league" not in df.columns:
            df["league"] = "UNKNOWN"

        if "pnl" not in df.columns:
            df["pnl"] = 0.0

        if "stake" not in df.columns:
            df["stake"] = 0.0

        if "ev" not in df.columns:
            df["ev"] = 0.0

        grouped = (
            df.groupby("league")
            .agg(
                bets=("league", "count"),
                total_pnl=("pnl", "sum"),
                total_stake=("stake", "sum"),
                avg_ev=("ev", "mean"),
            )
            .reset_index()
        )

        grouped["roi"] = grouped.apply(
            lambda row: (row["total_pnl"] / row["total_stake"]) if row["total_stake"] > 0 else 0.0,
            axis=1,
        )

        return grouped.to_dict("records")

from __future__ import annotations

import pandas as pd


class EdgeStabilityDetector:

    def analyze(self, rows: list[dict]) -> dict:
        if not rows:
            return {
                "count": 0,
                "avg_ev": 0.0,
                "std_ev": 0.0,
                "stable": False,
            }

        df = pd.DataFrame(rows).copy()

        if "ev" not in df.columns:
            df["ev"] = 0.0

        avg_ev = float(df["ev"].astype(float).mean())
        std_ev = float(df["ev"].astype(float).std(ddof=0))

        return {
            "count": int(len(df.index)),
            "avg_ev": round(avg_ev, 6),
            "std_ev": round(std_ev, 6),
            "stable": bool(std_ev <= max(0.05, abs(avg_ev))),
        }

from __future__ import annotations

import time

from quant.ops.heartbeat import Heartbeat
from quant.ops.monitoring import Monitoring


class Watchdog:

    def __init__(self, stale_after_seconds: int = 120):
        self.stale_after_seconds = int(stale_after_seconds)
        self.heartbeat = Heartbeat()
        self.monitoring = Monitoring()

    def ping(self, component: str):
        path = self.heartbeat.beat(component, "running")
        self.monitoring.log("watchdog_ping", {"component": component, "heartbeat_path": path})
        return path

    def loop_ping(self, component: str, interval_seconds: float = 30.0, cycles: int = 1):
        for _ in range(max(1, int(cycles))):
            self.ping(component)
            time.sleep(float(interval_seconds))

from __future__ import annotations

from quant.clv.bet_journal import BetJournal
from quant.live.closing_odds_collector import ClosingOddsCollector
from quant.live.edge_stability_detector import EdgeStabilityDetector
from quant.live.league_performance_analyzer import LeaguePerformanceAnalyzer
from quant.live.live_odds_scanner import LiveOddsScanner
from quant.live.market_optimizer import LeagueMarketPerformanceOptimizer
from quant.markets.market_expansion_runner import MarketExpansionRunner
from quant.ml.threshold_optimizer import ThresholdOptimizer
from quant.ops.heartbeat import Heartbeat
from quant.ops.monitoring import Monitoring


class FullSystemRunner:

    def __init__(self):
        self.market_runner = MarketExpansionRunner()
        self.live_scanner = LiveOddsScanner()
        self.league_analyzer = LeaguePerformanceAnalyzer()
        self.edge_detector = EdgeStabilityDetector()
        self.market_optimizer = LeagueMarketPerformanceOptimizer()
        self.threshold_optimizer = ThresholdOptimizer()
        self.closing_collector = ClosingOddsCollector()
        self.journal = BetJournal()
        self.heartbeat = Heartbeat()
        self.monitoring = Monitoring()

    def run_cycle(self, bankroll: float = 1000.0) -> dict:
        self.heartbeat.beat("full_system_runner", "running")
        self.monitoring.log("cycle_started", {"bankroll": bankroll})

        market_rows = self.market_runner.run(bankroll=bankroll)
        scanned_rows = self.live_scanner.scan(market_rows, bankroll=bankroll)

        threshold_summary = self.threshold_optimizer.optimize(scanned_rows)

        journal_rows = self.journal.list_bets()
        league_summary = self.league_analyzer.analyze(journal_rows)
        edge_summary = self.edge_detector.analyze(scanned_rows)
        market_summary = self.market_optimizer.optimize(journal_rows)

        payload = {
            "market_rows": market_rows,
            "scanned_rows": scanned_rows,
            "threshold_summary": threshold_summary,
            "league_summary": league_summary,
            "edge_summary": edge_summary,
            "market_summary": market_summary,
        }

        self.monitoring.log(
            "cycle_completed",
            {
                "market_rows": len(market_rows),
                "scanned_rows": len(scanned_rows),
            },
        )

        return payload
